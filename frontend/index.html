Quick start

Create an , then use these snippets to launch a session.

curl Create a session 
curl -X POST https://a4feg8qe30.execute-api.us-west-2.amazonaws.com/v1/sessions \
  -H "content-type: application/json" \
  -H "x-api-key: YOUR_BLIP_KEY" \
  -d '{
    "model": "claude-sonnet-4-6",
    "effort": "medium",
    "agent_options": {
      "allowedTools": ["Read", "Write", "Bash"],
      "permissionMode": "default"
    }
  }'
curl Bootstrap with your Claude key 
# Use the bootstrap_url from the session response
curl -X POST {bootstrap_url} \
  -H "content-type: application/json" \
  -d '{"api_key": "sk-ant-YOUR_CLAUDE_KEY"}'
curl Send a message 
curl -X POST https://a4feg8qe30.execute-api.us-west-2.amazonaws.com/v1/sessions/{session_id}/messages \
  -H "content-type: application/json" \
  -H "x-api-key: YOUR_BLIP_KEY" \
  -d '{"prompt": "What files are in this directory?"}'
python Full example 
import requests

BASE = "https://a4feg8qe30.execute-api.us-west-2.amazonaws.com"
BLIP_KEY = "YOUR_BLIP_KEY"
CLAUDE_KEY = "sk-ant-YOUR_CLAUDE_KEY"

headers = {"content-type": "application/json", "x-api-key": BLIP_KEY}

# 1. Create session
session = requests.post(f"{BASE}/v1/sessions", headers=headers, json={
    "model": "claude-sonnet-4-6",
    "effort": "medium",
    "agent_options": {"allowedTools": ["Read", "Bash"]}
}).json()

# 2. Bootstrap with Claude key
requests.post(session["bootstrap_url"], json={"api_key": CLAUDE_KEY})

# 3. Send message
sid = session["session_id"]
result = requests.post(f"{BASE}/v1/sessions/{sid}/messages", headers=headers,
    json={"prompt": "List files in the workspace"}
).json()

print(result)
node.js Full example 
const BASE = "https://a4feg8qe30.execute-api.us-west-2.amazonaws.com";
const BLIP_KEY = "YOUR_BLIP_KEY";
const CLAUDE_KEY = "sk-ant-YOUR_CLAUDE_KEY";

const headers = { "content-type": "application/json", "x-api-key": BLIP_KEY };

// 1. Create session
const session = await fetch(`${BASE}/v1/sessions`, {
  method: "POST", headers,
  body: JSON.stringify({
    model: "claude-sonnet-4-6",
    effort: "medium",
    agent_options: { allowedTools: ["Read", "Bash"] }
  })
}).then(r => r.json());

// 2. Bootstrap with Claude key
await fetch(session.bootstrap_url, {
  method: "POST",
  headers: { "content-type": "application/json" },
  body: JSON.stringify({ api_key: CLAUDE_KEY })
});

// 3. Send message
const result = await fetch(`${BASE}/v1/sessions/${session.session_id}/messages`, {
  method: "POST", headers,
  body: JSON.stringify({ prompt: "List files in the workspace" })
}).then(r => r.json());

console.log(result);
